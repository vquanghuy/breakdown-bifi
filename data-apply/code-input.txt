def read_file_to_string(file_path, chunk_size=1024):
    file_contents = ""
    with open(file_path, 'r', encoding='utf-8') as file:
        while True:
            chunk = file.read(chunk_size)
            if not chunk:
                break
            file_contents += chunk
    return file_contents


# Example usage:
file_path = 'your_file.txt'  # Replace with your file path
text = read_file_to_string(file_path)
print(text)
